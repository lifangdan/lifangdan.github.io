    # 2. 过滤树结构

    function filterTreeData(nodes,children, predicate){
  /**
   * 递归过滤节点，生成新的树结构
   * @param {Node[]} nodes 要过滤的节点
   * 树结构children字段
   * @param {node => boolean} predicate 过滤条件，符合条件的节点保留
   * @return 过滤后的节点
   */
  // 如果已经没有节点了，结束递归
  if (!(nodes && nodes.length)) {
    return [];
  }
  const newArr = [];
  for (const node of nodes) {
    if (predicate(node)) {
      // 如果节点符合条件，直接加入新的节点集
      newArr.push(node);
      node[children] = filterTreeData(node[children],children, predicate);
    } else {
      // 如果当前节点不符合条件，递归过滤子节点，
      // 把符合条件的子节点提升上来，并入新节点集
      newArr.push(...filterTreeData(node[children],children, predicate));
    }
  }
  // console.log('新的数组是：'+newArr)
  return newArr;
}

    # 六、深拷贝

    我们要拷贝数组或者对象，并对其进行一定的修改时，不希望原数组或对象也被修改，这时就需要用到深拷贝啦

    1.简单使用JSON.parse()和JSON.stringify()
let newArr=JSON.parse(JSON.stringify(Object))

2.
function deepClone(source) {
  if (!source || typeof source !== 'object') {
    return source
  }
  const targetObj = source.constructor === Array ? [] : {}
  Object.keys(source).forEach(keys => {
    if (source[keys] && typeof source[keys] === 'object') {
      targetObj[keys] = source[keys].constructor === Array ? [] : {}
      targetObj[keys] = deepClone(source[keys])
    } else {
      targetObj[keys] = source[keys]
    }
  })
  return targetObj
}

    # 1. 时间戳转换成日期时间

    function parseTime(time) {
  if (time) {
    var date = new Date(time)
    var year = date.getFullYear()
    /* 在日期格式中，月份是从0开始的，因此要加1
     * 使用三元表达式在小于10的前面加0，以达到格式统一  如 09:11:05
     * */
    var month = date.getMonth() + 1 < 10 ? '0' + (date.getMonth() + 1) : date.getMonth() + 1
    var day = date.getDate() < 10 ? '0' + date.getDate() : date.getDate()
    var hours = date.getHours() < 10 ? '0' + date.getHours() : date.getHours()
    var minutes = date.getMinutes() < 10 ? '0' + date.getMinutes() : date.getMinutes()
    var seconds = date.getSeconds() < 10 ? '0' + date.getSeconds() : date.getSeconds()
    return year + '-' + month + '-' + day + ' ' + hours + ':' + minutes + ':' + seconds
  } else {
    return ''
  }
}
